class Solution2 {
public:
    vector<int> twoSum(vector<int>& nums, int target)
    {
                map<int, int> m;
                for (int i = 0; i < nums.size(); i++) {
                        map<int, int>::iterator it = m.find(target - nums[i]);
                        if (it != m.end()) {
                                if (it->second != i) 
                                        return vector<int> {it->second, i};
                        } else {
                                m[nums[i]] = i;
                        }
                }
    }
};
